// [Code.gs] - 출석부 API 서버 (백엔드) function doPost(e) { try { const requestData = JSON.parse(e.postData.contents); const action = requestData.action; const settings = getSettings(); // G, H열의 설정값을 불러옴 // 1. 전화번호 목록 요청 시 if (action === "getAllPhones") { return createJsonResponse({ success: true, phones: getPhones() }); } // 2. 🔐 관리자 로그인 검증 요청 시 if (action === "verifyAdmin") { // 클라이언트가 보낸 비밀번호와 H2 셀의 비밀번호가 일치하는지 확인 if (requestData.password === settings["관리자비밀번호"]) { return createJsonResponse({ success: true }); } else { return createJsonResponse({ success: false, message: "비밀번호가 틀렸습니다." }); } } // 3. 🛡 출석 체크 요청 시 (보안 검증 추가) if (action === "markAttendance") { const phone = requestData.phoneNumber; // (A) 관리자 수동 출석인 경우: 관리자 비밀번호가 동봉되어야 함 if (requestData.adminPassword) { if (requestData.adminPassword !== settings["관리자비밀번호"]) { return createJsonResponse({ success: false, message: "🚨 서버 보안 차단: 관리자 권한이 없습니다." }); } } // (B) 학생 QR 스캔 출석인 경우: QR 토큰이 동봉되어야 함 else if (requestData.qrToken) { if (!validateQrToken(requestData.qrToken, settings["QR암호키"])) { return createJsonResponse({ success: false, message: "🚨 서버 보안 차단: 유효하지 않은 QR 코드입니다." }); } } // (C) 둘 다 없는 비정상적인 해킹 시도 else { return createJsonResponse({ success: false, message: "🚨 서버 보안 차단: 잘못된 접근입니다." }); } // 위 보안 검증을 무사히 통과한 경우에만 엑셀에 기록! const result = checkIn(phone); return createJsonResponse(result); } return createJsonResponse({ success: false, message: "알 수 없는 요청입니다." }); } catch (error) { return createJsonResponse({ success: false, message: "서버 오류: " + error.toString() }); } } // G, H열에서 설정값 불러오기 function getSettings() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("출석체크"); // G2부터 H20까지 넉넉하게 불러옴 const values = sheet.getRange("G2:H20").getValues(); const settings = {}; for (let i = 0; i < values.length; i++) { if (values[i][0]) { // G열에 값이 있으면 settings[String(values[i][0]).trim()] = String(values[i][1]).trim(); } } return settings; } // QR 토큰 검증 로직 (백엔드로 이동) function validateQrToken(token, secretKey) { try { // 프론트엔드에서 btoa로 인코딩한 값을 서버에서 디코딩 const decodedBlob = Utilities.newBlob(Utilities.base64Decode(token)); const decodedToken = decodedBlob.getDataAsString(); const parts = decodedToken.split('_'); const scannedDate = parts[0]; const scannedSecret = parts[1]; const today = new Date(); const todayString = Utilities.formatDate(today, Session.getScriptTimeZone(), 'yyyy-MM-dd'); // 날짜가 오늘이고, 암호키가 일치해야만 true 반환 if (scannedDate === todayString && scannedSecret === secretKey) { return true; } return false; } catch (e) { return false; } } function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } function getPhones() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("출석체크"); const values = sheet.getDataRange().getValues(); const phoneNumbers = []; for (let i = 1; i < values.length; i++) { if (values[i][2]) phoneNumbers.push(String(values[i][2])); } return phoneNumbers; } function checkIn(phoneNumber) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("출석체크"); const values = sheet.getDataRange().getValues(); const formattedTime = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss'); for (let i = 1; i < values.length; i++) { if (String(values[i][2]) === String(phoneNumber)) { sheet.getRange(i + 1, 4).setValue(formattedTime); return { success: true, studentId: values[i][0], name: values[i][1], time: formattedTime }; } } return { success: false, message: "등록되지 않은 전화번호입니다." }; }
